<template>
  <div class="w-full mb-[5px]">
    <div>
      <span class="font-bold">{{ hoggarinput.hogarDataLabels[props.cle] }}</span>
    </div>

    <div>
      <select
        class="w-full bg-[#E8E8E8] h-[50px] border-[darkblue] border-[1px]
        text-center"
        v-model="hoggarinput.hogarDataValues[props.cle]"
      >
        <option value="">-- Select --</option>
        <option
          v-for="(option, index) in hoggarinput.hogarDataOptions[props.cle]?.contents || []"
          :key="option"
          :value="option"
        >
          {{ hoggarinput.hogarDataOptions[props.cle]?.labels[index] || option }}
        </option>
      </select>
    </div>

    <div v-if="hoggarinput.errors[props.cle]" class="text-[red]">
      {{ hoggarinput.errors[props.cle] }}
    </div>
  </div>
</template>

<script setup>
import { watch } from 'vue'
import { HoggarInput } from '@/HoggarLibs/stores/hoggarinput'

const props = defineProps({
  cle: {
    type: String,
    required: true
  },
})

const hoggarinput = HoggarInput()

// Vérification au chargement et à chaque changement
watch(
  () => hoggarinput.hogarDataValues[props.cle],
  (newValue) => {
    const options = hoggarinput.hogarDataOptions[props.cle]?.contents || []
    if (!options.includes(newValue)) {
      hoggarinput.hogarDataValues[props.cle] = ''
    }
  },
  { immediate: true }
)
</script>
